CREATE TABLE reservas (id_reserva INT AUTO_INCREMENT NOT NULL, 
data_hora DATETIME NOT NULL, 
id_usuario INT NOT NULL, 
id_sala INT NOT NULL, 
hora VARCHAR(5) NOT NULL, 
status VARCHAR(1) NOT NULL, 
PRIMARY KEY(id_reserva)) DEFAULT CHARACTER SET utf8 COLLATE utf8_unicode_ci ENGINE = InnoDB;

CREATE TABLE salas (id_sala INT AUTO_INCREMENT NOT NULL, 
identificacao VARCHAR(255) NOT NULL, 
capacidade INT NOT NULL, 
PRIMARY KEY(id_sala)) DEFAULT CHARACTER SET utf8 COLLATE utf8_unicode_ci ENGINE = InnoDB;

CREATE TABLE usuarios (id_usuario INT AUTO_INCREMENT NOT NULL, 
nome VARCHAR(255) NOT NULL, 
usuario VARCHAR(255) NOT NULL, 
senha VARCHAR(255) NOT NULL, PRIMARY KEY(id_usuario)) DEFAULT CHARACTER SET utf8 COLLATE utf8_unicode_ci ENGINE = InnoDB;
